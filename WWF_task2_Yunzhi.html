Maize (Meal) Boxplot

Soybeans (Soymilk) Boxplot

Setup

library(ggplot2)
library(reshape2)

Read the dataset

DF<-read.csv("database.csv",header = F)
colnames(DF) = DF[3, ] # the third row will be the header

Maize (Meal) Boxplot

maize<-DF[277:428,66:76]
maize<-maize[,-2]
maize[] <- lapply(maize, function(x) as.numeric(as.character(x)))
ggplot(data = melt(maize), aes(x=variable, y=value, color=variable)) +
  geom_boxplot()+ 
  coord_flip()+
  ggtitle("Maize Boxplot")
